Below is one battle-tested pattern that ‚Äúhands the baton‚Äù from the splash star to the navbar in one uninterrupted motion.
It relies on two Framer-Motion tricks:

layoutId (shared-layout) ‚Äì Framer treats the same layoutId on two
different components as a single element that moves between layouts.
Animation controls ‚Äì we wait for the star‚Äôs 360¬∞ spin to finish and then
immediately let the navbar render, piggy-backing on the shared layout

// src/components/splash/Splash.tsx
"use client";

import { useEffect } from "react";
import { motion, useAnimation } from "framer-motion";
import { StarSpinner } from "@/components/ui/icons";
import { STAR_ID } from "@/lib/constants";

export default function Splash({ onDone }: { onDone: () => void }) {
  const controls = useAnimation();

  // run 1 full rotation, then call onDone()
  useEffect(() => {
    controls
      .start({
        rotate: 360,
        transition: { duration: 1.6, ease: "easeInOut" },
      })
      .then(() => onDone()); // triggers navbar render
  }, [controls, onDone]);

  return (
    <motion.div
      className="fixed inset-0 flex items-center justify-center
                 bg-gradient-to-br from-slate-900 via-blue-900 to-slate-800"
      initial={{ scale: 0.95, opacity: 0 }}
      animate={{ scale: 1,     opacity: 1 }}
      exit={{   scale: 0.95, opacity: 0, transition: { duration: 0.3 } }}
    >
      {/* star in centre */}
      <motion.div
        layoutId={STAR_ID}          // ü™Ñ  shared with navbar
        animate={controls}
      >
        <StarSpinner size={120} />
      </motion.div>
    </motion.div>
  );
}

"use client";

import Splash       from "@/components/splash/Splash";

  const [showSplash, setShowSplash] = useState(true);

  return (
    <>
      <AnimatePresence>
        {showSplash && (
          <Splash
            key="splash"
            onDone={() => setShowSplash(false)}
          />
        )}
      </AnimatePresence>

      {!showSplash && (

          <SiteHeader />
          <MainContent />
        </>
      )}
    </>
  );
